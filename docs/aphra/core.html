 49    def call_model(self, system_prompt, user_prompt, model_name, *,
 50                   log_call=False, enable_web_search=False,
 51                   web_search_context="high"):
 52        """
 53        Calls the model with the provided prompts.
 54
 55        :param system_prompt: The system prompt to set the context for the model.
 56        :param user_prompt: The user prompt to send to the model.
 57        :param model_name: The name of the model to use.
 58        :param log_call: Boolean indicating whether to log the call details.
 59        :param enable_web_search: Boolean indicating whether to enable web search via OpenRouter.
 60        :param web_search_context: Context size for web search ('low', 'medium', 'high').
 61        :return: The model's response.
 62        """
 63        response = None
 64        try:
 65            # Prepare the request parameters
 66            request_params = {
 67                "model": model_name,
 68                "messages": [
 69                    {"role": "system", "content": system_prompt},
 70                    {"role": "user", "content": user_prompt}
 71                ]
 72            }
 73
 74            # Add web search capabilities if enabled (OpenRouter format)
 75            if enable_web_search:
 76                # Append :online to model name for web search
 77                if not model_name.endswith(":online"):
 78                    request_params["model"] = f"{model_name}:online"
 79
 80                # Add web search options
 81                request_params["web_search_options"] = {
 82                    "search_context_size": web_search_context
 83                }
 84
 85            response = self.client.chat.completions.create(**request_params)
 86            response_content = response.choices[0].message.content
 87
 88            if log_call:
 89                self.log_model_call(user_prompt, response_content)
 90
 91            return response_content
 92        except requests.exceptions.RequestException as exc:
 93            logging.error('Request error: %s', exc)
 94            raise
 95        except (ValueError, KeyError, TypeError) as exc:
 96            logging.error('Error parsing response: %s', exc)
 97            if response and hasattr(response, 'text'):
 98                logging.error('Response content: %s', response.text)
 99            else:
100                logging.error('No response available')
101            raise
102
103    def log_model_call(self, user_prompt, response):
104        """
105        Logs the details of a model call to a log file.
106
107        :param user_prompt: The user prompt sent to the model.
108        :param response: The response received from the model.
109        """
110        if not self.logging_configured:
111            logging.basicConfig(filename='aphra.log', level=logging.INFO,
112                                format='%(asctime)s - %(levelname)s - %(message)s')
113            self.logging_configured = True
114
115        logging.info("\nUSER_PROMPT\n")
116        logging.info(user_prompt)
117        logging.info("\nRESPONSE\n")
118        logging.info(response)

13def parse_xml_tag(content: str, tag_name: str) -> Optional[str]:
14    """
15    Extract content from within XML-like tags in a string.
16
17    This is a generic parser that can extract content from any XML-like tag
18    in LLM responses, making it reusable across different workflows.
19
20    Args:
21        content: The string content containing XML-like tags
22        tag_name: The name of the tag to extract (without < >)
23
24    Returns:
25        str: The content within the tags, or None if not found
26
27    Example:
28        >>> content = "Some text <result>Hello World</result> more text"
29        >>> parse_xml_tag(content, "result")
30        "Hello World"
31    """
32    try:
33        start_tag = f"<{tag_name}>"
34        end_tag = f"</{tag_name}>"
35
36        start_index = content.find(start_tag)
37        if start_index == -1:
38            logging.warning("Start tag '<%s>' not found in content", tag_name)
39            return None
40
41        start_index += len(start_tag)
42        end_index = content.find(end_tag, start_index)
43
44        if end_index == -1:
45            logging.warning("End tag '</%s>' not found in content", tag_name)
46            return None
47
48        extracted_content = content[start_index:end_index].strip()
49        return extracted_content
50
51    except Exception as exc:
52        logging.error("Error parsing XML tag '%s': %s", tag_name, exc)
53        return None

55def parse_multiple_xml_tags(content: str, tag_name: str) -> list[str]:
56    """
57    Extract content from multiple XML-like tags of the same type.
58
59    Args:
60        content: The string content containing XML-like tags
61        tag_name: The name of the tag to extract (without < >)
62
63    Returns:
64        list[str]: List of content within all matching tags
65
66    Example:
67        >>> content = "Text <item>First</item> more <item>Second</item> end"
68        >>> parse_multiple_xml_tags(content, "item")
69        ["First", "Second"]
70    """
71    try:
72        # Use regex to find all occurrences
73        pattern = f"<{re.escape(tag_name)}>(.*?)</{re.escape(tag_name)}>"
74        matches = re.findall(pattern, content, re.DOTALL)
75
76        # Strip whitespace from each match
77        results = [match.strip() for match in matches]
78        return results
79
80    except Exception as exc:
81        logging.error("Error parsing multiple XML tags '%s': %s", tag_name, exc)
82        return []

def parse_xml_tag_with_attributes(content: str, tag_name: str) -> Optional[dict]: 
 84def parse_xml_tag_with_attributes(content: str, tag_name: str) -> Optional[dict]:
 85    """
 86    Extract content and attributes from XML-like tags.
 87
 88    Args:
 89        content: The string content containing XML-like tags
 90        tag_name: The name of the tag to extract (without < >)
 91
 92    Returns:
 93        dict: Dictionary with 'content' and 'attributes' keys, or None if not found
 94
 95    Example:
 96        >>> content = 'Text <result type="success">Hello World</result>'
 97        >>> parse_xml_tag_with_attributes(content, "result")
 98        {"content": "Hello World", "attributes": {"type": "success"}}
 99    """
100    try:
101        # Pattern to match tag with optional attributes
102        pattern = f"<{re.escape(tag_name)}([^>]*)>(.*?)</{re.escape(tag_name)}>"
103        match = re.search(pattern, content, re.DOTALL)
104
105        if not match:
106            logging.warning("Tag '<%s>' not found in content", tag_name)
107            return None
108
109        attributes_str = match.group(1).strip()
110        tag_content = match.group(2).strip()
111
112        # Parse attributes if any
113        attributes = {}
114        if attributes_str:
115            # Simple attribute parsing (handles key="value" format)
116            attr_pattern = r'(\w+)="([^"]*)"'
117            attributes = dict(re.findall(attr_pattern, attributes_str))
118
119        return {
120            'content': tag_content,
121            'attributes': attributes
122        }
123
124    except Exception as exc:
125        logging.error("Error parsing XML tag with attributes '%s': %s", tag_name, exc)
126        return None
