 29    def load_config(self, config_file_path):
 30        """
 31        Loads configuration from a TOML file.
 32
 33        :param config_file_path: Path to the TOML file.

 35        try:
 36            with open(config_file_path, 'r', encoding='utf-8') as file:

 38            self.api_key_openrouter = config['openrouter']['api_key']

 40        except FileNotFoundError:
 41            logging.error('File not found: %s', config_file_path)
 42            raise
 43        except toml.TomlDecodeError:

 46        except KeyError as e:
 47            logging.error('Missing key in config file: %s', e)

 50    def call_model(self, system_prompt, user_prompt, model_name, log_call=False, 
 51                   enable_web_search=False, web_search_context="high"):
 52        """
 53        Calls the model with the provided prompts.
 54
 55        :param system_prompt: The system prompt to set the context for the model.
 56        :param user_prompt: The user prompt to send to the model.
 57        :param model_name: The name of the model to use.
 58        :param log_call: Boolean indicating whether to log the call details.
 59        :param enable_web_search: Boolean indicating whether to enable web search via OpenRouter.
 60        :param web_search_context: Context size for web search ('low', 'medium', 'high').
 61        :return: The model's response.
 62        """
 63        response = None
 64        try:
 65            # Prepare the request parameters
 66            request_params = {
 67                "model": model_name,
 68                "messages": [
 69                    {"role": "system", "content": system_prompt},
 70                    {"role": "user", "content": user_prompt}
 71                ]
 72            }
 73            
 74            # Add web search capabilities if enabled (OpenRouter format)
 75            if enable_web_search:
 76                # Append :online to model name for web search
 77                if not model_name.endswith(":online"):
 78                    request_params["model"] = f"{model_name}:online"
 79                
 80                # Add web search options
 81                request_params["web_search_options"] = {
 82                    "search_context_size": web_search_context
 83                }
 84
 85            response = self.client.chat.completions.create(**request_params)
 86            response_content = response.choices[0].message.content
 87
 88            if log_call:
 89                self.log_model_call(user_prompt, response_content)
 90
 91            return response_content
 92        except requests.exceptions.RequestException as e:
 93            logging.error('Request error: %s', e)
 94            raise
 95        except (ValueError, KeyError, TypeError) as e:
 96            logging.error('Error parsing response: %s', e)
 97            if response and hasattr(response, 'text'):
 98                logging.error('Response content: %s', response.text)
 99            else:
100                logging.error('No response available')
101            raise
102
103    def log_model_call(self, user_prompt, response):
104        """
105        Logs the details of a model call to a log file.
106
107        :param user_prompt: The user prompt sent to the model.
108        :param response: The response received from the model.
109        """
110        if not self.logging_configured:
111            logging.basicConfig(filename='aphra.log', level=logging.INFO,
112                                format='%(asctime)s - %(levelname)s - %(message)s')
113            self.logging_configured = True
114
115        logging.info("\nUSER_PROMPT\n")
116        logging.info(user_prompt)
117        logging.info("\nRESPONSE\n")
118        logging.info(response)

 9def parse_analysis(analysis_str):
10    """
11    Parses the analysis part of the provided string and returns
12    a list of items with their names and keywords.
13
14    :param analysis_str: String containing the analysis in the specified format.
15    :return: A list of dictionaries, each containing 'name' and 'keywords' from the analysis.
16    """
17    try:
18        # Extract the <analysis> part
19        analysis_start = analysis_str.index("<analysis>") + len("<analysis>")
20        analysis_end = analysis_str.index("</analysis>")
21        analysis_content = analysis_str[analysis_start:analysis_end].strip()
22
23        # Parse the analysis content using XML parser
24        root = ET.fromstring(f"<root>{analysis_content}</root>")
25        items = []
26
27        for item in root.findall('item'):
28            name = item.find('name').text
29            keywords = item.find('keywords').text
30            items.append({'name': name, 'keywords': keywords.split(', ')})
31
32        return items
33    except ValueError as e:
34        logging.error('Error parsing analysis string: %s', e)
35        return []
36    except ET.ParseError as e:
37        logging.error('Error parsing XML content: %s', e)
38        return []

 14class WorkflowRegistry:
 15    """
 16    Registry for managing translation workflows.
 17    
 18    This class maintains a registry of available workflows and provides
 19    methods for workflow discovery and selection.
 20    """
 21    
 22    def __init__(self):
 23        """Initialize the workflow registry with default workflows."""
 24        self._workflows: Dict[str, Type[AbstractWorkflow]] = {}
 25        self._register_default_workflows()
 26    
 27    def _register_default_workflows(self):
 28        """Register the default workflows that come with Aphra."""
 29        self.register_workflow(ArticleWorkflow)
 30    
 31    def register_workflow(self, workflow_class: Type[AbstractWorkflow]):
 32        """
 33        Register a new workflow type.
 34        
 35        Args:
 36            workflow_class: The workflow class to register
 37        """
 38        # Create temporary instance to get the workflow name
 39        temp_workflow = workflow_class()
 40        workflow_name = temp_workflow.get_workflow_name()
 41        self._workflows[workflow_name] = workflow_class
 42    
 43    def get_workflow(self, workflow_name: str) -> Optional[AbstractWorkflow]:
 44        """
 45        Get a workflow instance by name.
 46        
 47        Args:
 48            workflow_name: The name of the workflow to retrieve
 49            
 50        Returns:
 51            AbstractWorkflow: An instance of the requested workflow, or None if not found
 52        """
 53        workflow_class = self._workflows.get(workflow_name)
 54        if workflow_class:
 55            return workflow_class()
 56        return None
 57    
 58    def get_suitable_workflow(self, text: str, **kwargs) -> Optional[AbstractWorkflow]:
 59        """
 60        Find the most suitable workflow for the given content.
 61        
 62        Args:
 63            text: The text content to analyze
 64            **kwargs: Additional parameters for workflow evaluation
 65            
 66        Returns:
 67            AbstractWorkflow: The most suitable workflow instance, or None if none found
 68        """
 69        # For now, we check workflows in registration order
 70        # In the future, we could implement more sophisticated selection logic
 71        for workflow_class in self._workflows.values():
 72            workflow = workflow_class()
 73            if workflow.is_suitable_for(text, **kwargs):
 74                return workflow
 75        
 76        return None
 77    
 78    def list_workflows(self) -> List[str]:
 79        """
 80        Get a list of all registered workflow names.
 81        
 82        Returns:
 83            List[str]: Names of all registered workflows
 84        """
 85        return list(self._workflows.keys())
 86    
 87    def get_workflow_info(self, workflow_name: str) -> Optional[Dict[str, str]]:
 88        """
 89        Get information about a specific workflow.
 90        
 91        Args:
 92            workflow_name: The name of the workflow
 93            
 94        Returns:
 95            Dict[str, str]: Information about the workflow, or None if not found
 96        """
 97        workflow = self.get_workflow(workflow_name)
 98        if workflow:
 99            return {
100                'name': workflow.get_workflow_name(),
101                'class': workflow.__class__.__name__,
102                'module': workflow.__class__.__module__
103            }
104        return None
